113  void CircularBuffer::push(float* _input){
114 
115  csize += 1;
116 
117  float* w_i1 = &data[0];
118 
119  // updating the window queue
120  if (csize >= maxSize){
121 
122  csize = maxSize;
123  float* w_i2 = w_i1 + nDof;
124 
125  // dequeuing the last element
126  for (int i = nDof ; i < dataSize; i++, w_i1++, w_i2++){
127  *w_i1 = *w_i2;
128  }
129  }else{
130  w_i1 += csize*nDof;
131 
132  }
133 
134  // queuing the newly generated target
135  float* w_i2 = &_input[0];
136  for (int i = 0 ; i < nDof; i++, w_i1++, w_i2++){
137  *w_i1 = *w_i2;
138  }
139  }
