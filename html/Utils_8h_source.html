305  template <typename T>
306  inline void Utils::adam(T* _p, T* _g, T* _m, T* _v, int _epoch, float _alpha, float _beta1, float _beta2){
307  auto p = _p->data();
308  auto g = _g->data();
309  auto m = _m->data();
310  auto v = _v->data();
311  for (int i = 0 ; i < _p->size(); i++, p++, g++, m++, v++){
312  *m = _beta1*(*m) + (1.0-_beta1)*(*g);
313  *v = _beta2*(*v) + (1.0-_beta2)*((*g) * (*g));
314  float mHat = *m / (1.0 - pow(_beta1, _epoch));
315  float vHat = *v / (1.0 - pow(_beta2, _epoch));
316  *p -= _alpha *mHat/(pow(vHat, 0.5)+NON_ZERO);
317  }
318  }
