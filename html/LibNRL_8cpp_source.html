111  void LibNRL::newModel(string path){
112 
113  deallocate();
114 
115  propPath = path;
116 
117  map<string,float1DContainer> float1DMap;
118  map<string,string> stringMap;
119  map<string,bool> boolMap;
120 
121  loaded = false;
122  dsoft = 10;
123  sigma = 0.2;
124 
125  t_retrain = false;
126  t_nEpoch = 0;
127  t_greedy = false;
128  t_beta1 = 0.9;
129  t_beta2 = 0.999;
130  t_alpha = 0.001;
131  t_step=1;
132 
133  e_nEpoch = 0;
134  e_step = 1;
135  e_beta1 = 0.9;
136  e_beta2 = 0.999;
137  e_alpha = 0.1;
138  e_winSize = 0;
139 
140  maxLoss = std::numeric_limits<float>::max();
141  logFile = 0;
142  stdoutLog = false;
143 
144  try{
145 
146  ut->getProperties(float1DMap, stringMap, boolMap, propPath);
147 
148  if(stringMap.find("modelpath") == stringMap.end()) throw Exception("'modelpath' property not found");
149  modelPath = stringMap["modelpath"];
150 
151  if(stringMap.find("datapath") == stringMap.end()) throw Exception("'datapath' property not found");
152  dataPath = stringMap["datapath"];
153 
154  if(stringMap.find("network") == stringMap.end()) throw Exception("'network' property not found");
155  networkName = stringMap["network"];
156 
157  if(stringMap.find("robot") == stringMap.end()) throw Exception("'robot' property not found");
158  robotName = stringMap["robot"];
159 
160  if(boolMap.find("shuffle") == boolMap.end()) throw Exception("'shuffle' property not found");
161  t_shuffle = boolMap["shuffle"];
162 
163  if(boolMap.find("retrain") == boolMap.end()) throw Exception("'retrain' property not found");
164  t_retrain = boolMap["retrain"];
165 
166  if(boolMap.find("greedy") == boolMap.end()) throw Exception("'greedy' property not found");
167  t_greedy = boolMap["greedy"];
168 
169  if(float1DMap.find("epochs") == float1DMap.end()) throw Exception("'epochs' property not found");
170  t_nEpoch = int(float1DMap["epochs"][0]);
171 
172  if(float1DMap.find("dsoft") == float1DMap.end()) throw Exception("'dsoft' property not found");
173  dsoft = float1DMap["dsoft"][0];
174 
175  if(float1DMap.find("sigma") == float1DMap.end()) throw Exception("'sigma' property not found");
176  sigma = float1DMap["sigma"][0];
177 
178  if(float1DMap.find("beta1") == float1DMap.end()) throw Exception("'beta1' property not found");
179  t_beta1 = float1DMap["beta1"][0];
180 
181  if(float1DMap.find("beta2") == float1DMap.end()) throw Exception("'beta2' property not found");
182  t_beta2 = float1DMap["beta2"][0];
183 
184  if(float1DMap.find("alpha") == float1DMap.end()) throw Exception("'alpha' property not found");
185  t_alpha = float1DMap["alpha"][0];
186 
187  if(float1DMap.find("activejoints") == float1DMap.end()) throw Exception("'activejoints' property not found");
188  float1DContainer fActiveJoints = float1DMap["activejoints"];
189  for (unsigned int i = 0; i < fActiveJoints.size(); i++)
190  activeJoints.push_back(int(fActiveJoints[i])==1);
191 
192  if(float1DMap.find("nsamples") == float1DMap.end()) throw Exception("'nsamples' property not found");
193  float1DContainer fSamples = float1DMap["nsamples"];
194  for (unsigned int i = 0; i < fSamples.size(); i++)
195  nSamples.push_back(int(fSamples[i]));
196 
197  stringstream stream;
198  stream << modelPath << "/epoch.d";
199  strEpoch = stream.str();
200 
201  if (t_retrain){
202  ifstream eFile(strEpoch);
203  if (!eFile.is_open()){
204  cout << "Warning: the file [" << strEpoch << "] is unavailable, retrain set false";
205  t_retrain = false;
206  }
207  else{
208  float1DContainer vec;
209  ut->loadContainer<float1DContainer>(&eFile, &vec, ut->getDelimiter());
210  t_step = int(vec[0]+1);
211  maxLoss = vec[1];
212  }
213  }
214 
215  if (robotName == "torobo")
216  robot = Torobo::getInstance(activeJoints);
217  else if (robotName == "cartesian")
218  robot = Cartesian::getInstance(activeJoints);
219  else if (robotName == "generic")
220  robot = Generic::getInstance(activeJoints);
221  else{
222  stringstream stream;
223  stream << "unknown 'robot' property [" << robotName << "]";
224  throw Exception(stream.str());
225  }
226 
227  dataset = new oist::Dataset(dataPath, dataPrefix, nSamples, dsoft, sigma, ut, robot);
228  dataset->encodeSoftmax(YSoftmax);
229 
230  nSeq = nSamples.size();
231 
232  t_prim_Ids.clear();
233  for (int i = 0; i < nSeq; i++)
234  t_prim_Ids.push_back(i);
235 
236  nDof = ((float)robot->getDOF())*1.0;
237  seqLen = dataset->getPrimLength();
238 
239  if (networkName == "pvrnn")
240  model = new NetworkPvrnn(float1DMap, dataset);
241  else if (networkName == "pvrnnbeta")
242  model = new NetworkPvrnnBeta(float1DMap, dataset);
243  else{
244  stringstream stream;
245  stream << "unknown 'network' property [" << networkName << "]";
246  throw Exception(stream.str());
247  }
248 
249 
250  }catch(Exception& _e){
251  cout << "Error: " << _e.what() << endl;
252  deallocate();
253  }catch(...){
254  cout << "Error: unknown exception when extracting data from properties "<< endl;
255  deallocate();
256  }

359  void LibNRL::t_loop(float* output, int n){
360 
361  if (model == nullptr){
362  cout << modelNUllMsg << endl;
363  return;
364  }
365  bool keepRunning = true;
366 
367  t_nEpoch = t_step + n - 1;
368  if (keepRunning){
369  float loss = 0.0;
370  float reconstruction = 0.0;
371  float regulation = 0.0;
372 
373  chrono::high_resolution_clock::time_point mst1 = chrono::high_resolution_clock::now();
374 
375  for (; t_step <= t_nEpoch; t_step++){
376  vectorXf3DContainer All_X;
377 
378  if (t_shuffle)
379  ut->shuffle<int1DContainer>(&t_prim_Ids);
380 
381  int1DContainer::iterator t_prim_Ids_i = t_prim_Ids.begin();
382  for (int pId = 0; pId < nSeq; pId++, t_prim_Ids_i++){
383  vectorXf2DContainer X;
384  model->t_forward(seqLen, *t_prim_Ids_i, X);
385  All_X.push_back(X);
386  }
387 
388  t_prim_Ids_i = t_prim_Ids.begin();
389  loss = 0.0;
390  reconstruction = 0.0;
391  regulation = 0.0;
392  for (int pId = 0; pId < nSeq; pId++, t_prim_Ids_i++){
393  vectorXf3DContainer Y_p = YSoftmax[*t_prim_Ids_i];
394  vectorXf2DContainer X = All_X[pId];
395  model->t_backward(*t_prim_Ids_i, X, Y_p, reconstruction, regulation, loss);
396  }
397  model->t_optAdam(t_step, t_alpha, t_beta1, t_beta2);
398 
399  if (t_step % n == 0){
400  float mseGen = 0.0;
401  for (int pId = 0; pId < nSeq; pId++){
402  vectorXf2DContainer X;
403  vectorXf3DContainer Y_p = YSoftmax[pId];
404  model->t_generate(seqLen, pId, X);
405  mseGen += model->getRecError(X, Y_p);
406  }
407 
408  chrono::high_resolution_clock::time_point mst2 = chrono::high_resolution_clock::now();
409  chrono::duration<double, std::milli> msdiff = mst2 - mst1;
410 
411  float oTime = (float)msdiff.count();
412 
413  if (stdoutLog == true){
414  cout << "Epoch ["<< t_step <<
415  "] - Time ["<< oTime <<
416  "ms] - RE_Q ["<< reconstruction <<
417  "] - RE_P ["<< mseGen <<
418  "] - Regulation ["<< regulation <<
419  "] - loss [" << loss << "]" << endl;
420  }
421 
422  *logFile<< "Epoch ["<< t_step <<
423  "] - Time ["<< oTime <<
424  "ms] - RE_Q ["<< reconstruction <<
425  "] - RE_P ["<< mseGen <<
426  "] - Regulation ["<< regulation <<
427  "] - loss [" << loss << "]" << endl;
428 
429  output[0] = t_step;
430  output[1] = oTime;
431  output[2] = reconstruction;
432  output[3] = mseGen;
433  output[4] = regulation;
434  output[5] = loss;
435  output[6] = 0.0;
436 
437  if (maxLoss > loss || !t_greedy){
438  maxLoss = loss;
439  model->save(modelPath);
440  output[6] = 1.0;
441  ofstream eFile(strEpoch,std::ofstream::out);
442  if (eFile.is_open()){
443  eFile << t_step << ut->getDelimiter() << loss;
444  eFile.close();
445  }
446  *logFile<< "The model has been saved" << endl;
447  }
448  mst1 = chrono::high_resolution_clock::now();
449  }
450 
451  }
452  }

641  void LibNRL::e_postdict(float* input, float* output, bool show){
642  if (model == nullptr){
643  cout << modelNUllMsg << endl;
644  return;
645  }
646  if (loaded == false){
647  cout << "Warning: The model should be loaded before calling e_postdict!" << endl;
648  return;
649  }
650 
651  vectorXf2DContainer Y;
652  int size []= {e_winSize, (int)nDof};
653 
654  dataset->encodeSoftmax(input, size, Y);
655 
656  float maxLoss = std::numeric_limits<float>::max();
657 
658  float loss = 0.0;
659  float rec = 0.0;
660  float reg = 0.0;
661 
662  if(model->e_initForward()){
663 
664  for (int e1 = 1 ; e1 <= e_nEpoch; e1++, e_step++){
665  loss = 0.0;
666  rec = 0.0;
667  reg = 0.0;
668 
669  vectorXf2DContainer X;
670  model->e_forward(X);
671  model->e_backward(X, Y, rec, reg, loss);
672  if (show)
673  cout << "E[" << e_step << "]" << " REC[" << rec << "] " << " REG[" << reg << "] loss[" << loss << "]" << endl;
674  model->e_optAdam(e_step, e_alpha, e_beta1, e_beta2);
675 
676  if (maxLoss > loss){
677  output[0] = loss;
678  output[1] = rec;
679  output[2] = reg;
680  maxLoss = loss;
681  model->e_copyParam();
682  }
683  }
684  model->e_overwriteParam();
685  }
