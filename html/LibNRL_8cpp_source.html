347  void LibNRL::t_loop(float* output, int n){
348 
349  if (model == nullptr){
350  cout << modelNUllMsg << endl;
351  return;
352  }
353  bool keepRunning = true;
354 
355  t_nEpoch = t_step + n - 1;
356  if (keepRunning){
357  float loss = 0.0;
358  float reconstruction = 0.0;
359  float regulation = 0.0;
360 
361  chrono::high_resolution_clock::time_point mst1 = chrono::high_resolution_clock::now();
362 
363  for (; t_step <= t_nEpoch; t_step++){
364  vectorXf3DContainer All_X;
365 
366  if (t_shuffle)
367  ut->shuffle<int1DContainer>(&t_prim_Ids);
368 
369  int1DContainer::iterator t_prim_Ids_i = t_prim_Ids.begin();
370  for (int pId = 0; pId < nSeq; pId++, t_prim_Ids_i++){
371  vectorXf2DContainer X;
372  model->t_forward(seqLen, *t_prim_Ids_i, X);
373  All_X.push_back(X);
374  }
375 
376  t_prim_Ids_i = t_prim_Ids.begin();
377  loss = 0.0;
378  reconstruction = 0.0;
379  regulation = 0.0;
380  for (int pId = 0; pId < nSeq; pId++, t_prim_Ids_i++){
381  vectorXf3DContainer Y_p = YSoftmax[*t_prim_Ids_i];
382  vectorXf2DContainer X = All_X[pId];
383  model->t_backward(*t_prim_Ids_i, X, Y_p, reconstruction, regulation, loss);
384  }
385  model->t_optAdam(t_step, t_alpha, t_beta1, t_beta2);
386 
387  if (t_step % n == 0){
388  float mseGen = 0.0;
389  for (int pId = 0; pId < nSeq; pId++){
390  vectorXf2DContainer X;
391  vectorXf3DContainer Y_p = YSoftmax[pId];
392  model->t_generate(seqLen, pId, X);
393  mseGen += model->getRecError(X, Y_p);
394  }
395 
396  chrono::high_resolution_clock::time_point mst2 = chrono::high_resolution_clock::now();
397  chrono::duration<double, std::milli> msdiff = mst2 - mst1;
398 
399  float oTime = (float)msdiff.count();
400 
401  if (stdoutLog == true){
402  cout << "Epoch ["<< t_step <<
403  "] - Time ["<< oTime <<
404  "ms] - RE_Q ["<< reconstruction <<
405  "] - RE_P ["<< mseGen <<
406  "] - Regulation ["<< regulation <<
407  "] - loss [" << loss << "]" << endl;
408  }
409 
410  *logFile<< "Epoch ["<< t_step <<
411  "] - Time ["<< oTime <<
412  "ms] - RE_Q ["<< reconstruction <<
413  "] - RE_P ["<< mseGen <<
414  "] - Regulation ["<< regulation <<
415  "] - loss [" << loss << "]" << endl;
416 
417  output[0] = t_step;
418  output[1] = oTime;
419  output[2] = reconstruction;
420  output[3] = mseGen;
421  output[4] = regulation;
422  output[5] = loss;
423  output[6] = 0.0;
424 
425  if (maxLoss > loss || !t_greedy){
426  maxLoss = loss;
427  model->save(modelPath);
428  output[6] = 1.0;
429  ofstream eFile(strEpoch,std::ofstream::out);
430  if (eFile.is_open()){
431  eFile << t_step << " " << loss;
432  eFile.close();
433  }
434  *logFile<< "The model has been saved" << endl;
435  }
436  mst1 = chrono::high_resolution_clock::now();
437  }
438 
439  }
440  }
