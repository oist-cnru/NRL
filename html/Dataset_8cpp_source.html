55 
56  // setting path and number os samples

58  path = _path;
59  dataPrefix = _dataPrefix;
60  dsoft = _dsoft;

62  nPrims = _samples.size();
63  copy(_samples.begin(), _samples.end(), back_inserter(nSamples));
64

68  _robot->getJointRange(jrange);
69 
70  // setting the encoding resolution
71  encDim = 0;
72  seqLen = 0;
73 
74  for (int j = 0; j< nDof; j ++){
75  int r = (int)ceil(jrange[j]/dsoft);
76  encDim += r;
77  nUnits.push_back(r);

81  for (int i = 0; i < r ; i++ ){
82  ref_i.push_back(ri);
83  ri += linStep;
84  }
85  ref.push_back(ref_i);

154 void Dataset::encodeSoftmax(float* input, int* size, vectorXf2DContainer& output){
155 
156  float sigma2 = sigma*sigma;
157  int nT = size[0];
158  int nDof = size[1];
159 
160  for (int t = 0; t < nT; t++){
161 
162  vectorXf1DContainer encData_t;
163  float* d_t = input+(t*nDof);
164 
165  float2DContainer::iterator ref_ = ref.begin();
166  float1DContainer::iterator jmin_ = jmin.begin();
167  float1DContainer::iterator jrange_ = jrange.begin();
168  int1DContainer::iterator encU_ = nUnits.begin();
169 
170  for (int j = 0; j < nDof; j++, ref_++, jmin_++, jrange_++, encU_++){
171 
172  float d_tj = *(d_t+j);
173  float v = ((d_tj - (*jmin_))/(*jrange_)) + NON_ZERO;
174  int encUnits = *encU_;
175 
176  float normalization = 0.0;
177  float1DContainer::iterator ref_j = ref_->begin();
178  VectorXf encPSJ = VectorXf::Zero(encUnits);
179  auto dv_ = encPSJ.data();
180  for (int r = 0; r < encUnits; r++, ref_j++, dv_++){
181  float softmax = exp(-(pow((*ref_j)- v, 2.0))/sigma2);
182  *dv_ = softmax;
183  normalization += softmax;
184  }
185 
186  encPSJ/= normalization;
187  encData_t.push_back(encPSJ);
188 
189  }
190  output.push_back(encData_t);
191  }
192 }

194 void Dataset::softmax(float2DContainer& input, vectorXf2DContainer& output){
195 
196  float sigma2 = sigma*sigma;
197  int nT = input.size();
198 
199  for (int t = 0; t < nT; t++){
200 
201  float1DContainer d_t = input[t];
202  vectorXf1DContainer encUnits_j;
203 
204  for (int j = 0; j < nDof; j++){
205 
206  int encUnits = nUnits[j];
207 
208  VectorXf encPSJ = VectorXf::Zero(encUnits);
209 
210  float d_tj = d_t[j];
211 
212  vector<float> ref_j = ref[j];
213 
214  float v = (d_tj - jmin[j])/jrange[j];
215 
216  float normalization = 0.0;
217 
218  for (int r = 0; r < encUnits; r++){
219 
220  float softmax = exp(-(pow(ref_j[r]- v, 2.0))/sigma2);
221  encPSJ(r) = softmax;
222  normalization += softmax;
223  }
224 
225  // normalizing the distributions
226  for (int r = 0; r < encUnits; r++){
227  encPSJ(r) /= normalization;
228  }
229  encUnits_j.push_back(encPSJ);
230 
231  }
232  output.push_back(encUnits_j);
233  }
