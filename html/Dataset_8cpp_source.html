53 Dataset::Dataset(string _path, string _dataPrefix, int1DContainer& _samples, float _dsoft, float _sigma, Utils* _ut, IRobot* _robot){
54 
55  // setting path and number os samples
56  ut = _ut;
57  path = _path;
58  dataPrefix = _dataPrefix;
59  dsoft = _dsoft;
60  sigma = _sigma;
61  nPrims = _samples.size();
62  copy(_samples.begin(), _samples.end(), back_inserter(nSamples));
63 
64  nDof = _robot->getDOF();
65  _robot->getJointHighLimit(jmax);
66  _robot->getJointLowLimit(jmin);
67  _robot->getJointRange(jrange);
68 
69  // setting the encoding resolution
70  encDim = 0;
71  seqLen = 0;
72 
73  for (int j = 0; j< nDof; j ++){
74  int r = (int)ceil(jrange[j]/dsoft);
75  encDim += r;
76  nUnits.push_back(r);
77  float ri = 0.0;
78  float linStep = 1.0/((float)((r-1)*1.0));
79  float1DContainer ref_i;
80  for (int i = 0; i < r ; i++ ){
81  ref_i.push_back(ri);
82  ri += linStep;
83  }
84  ref.push_back(ref_i);
85  }

89 void Dataset::loadData(string _path, string _dataPrefix, int _nPrims, int1DContainer _nSamples, float4DContainer& _data){
90 
91  for (int p = 0; p < _nPrims; p++){
92  float3DContainer d_p;
93  for (int s = 0; s < nSamples[p]; s++){
94  stringstream stream;
95  stream << _path << "/" << dataPrefix << "_" << p << "_" << s << ".csv";
96  std::ifstream file(stream.str());
97  if (!file.good()){
98  stringstream stream2;
99  stream2 << "The file [" << stream.str() << "] could not be opened. Hint: check if the parameter 'nsamples' in the model property file corresponds to the dataset samples'" << endl;
100  throw Exception(stream2.str());
101  }
102  std::string line = "";
103  float2DContainer d_ps;
104 
105  // Iterate through each line and split the content using delimiter
106  int nLine = 0;
107  while (getline(file, line)){
108  nLine ++;
109  float1DContainer d_pst;
110 
111  std::vector<string> vec;
112  ut->split(vec, line);
113 
114  if (nDof > (int)vec.size()){
115  stringstream stream;
116  stream << "Error: Please check the data row delimiter '" << ut->getDelimiter() << "'. Its was obtained less data that available Degrees of Freedom at line #" << nLine << endl;
117  throw Exception(stream.str());
118  }
119  for (std::vector<string>::iterator j = vec.begin() ; j != vec.end(); j++)
120  d_pst.push_back(stof(*j));
121  d_ps.push_back(d_pst);
122  }
123  if ((int)d_ps.size() > seqLen)
124  seqLen = d_ps.size();
125 
126  d_p.push_back(d_ps);
127  file.close();
128  }
129  _data.push_back(d_p);
130  }
131  if (seqLen == 0){
132  stringstream stream;
133  stream << "A sequence with length zero was found. Please check the dataset directory path. Alternatively, make sure all the sequences were recorded!" << endl;
134  throw Exception(stream.str());
135  }
136 
137  cout << "Dataset loaded. Primitive number: " << _data.size() << ", length: " << seqLen << " steps" << endl;
138 
139 }

153 void Dataset::encodeSoftmax(float* input, int* size, vectorXf2DContainer& output){
154 
155  float sigma2 = sigma*sigma;
156  int nT = size[0];
157  int nDof = size[1];
158 
159  for (int t = 0; t < nT; t++){
160 
161  vectorXf1DContainer encData_t;
162  float* d_t = input+(t*nDof);
163 
164  float2DContainer::iterator ref_ = ref.begin();
165  float1DContainer::iterator jmin_ = jmin.begin();
166  float1DContainer::iterator jrange_ = jrange.begin();
167  int1DContainer::iterator encU_ = nUnits.begin();
168 
169  for (int j = 0; j < nDof; j++, ref_++, jmin_++, jrange_++, encU_++){
170 
171  float d_tj = *(d_t+j);
172  float v = ((d_tj - (*jmin_))/(*jrange_)) + NON_ZERO;
173  int encUnits = *encU_;
174 
175  float normalization = 0.0;
176  float1DContainer::iterator ref_j = ref_->begin();
177  VectorXf encPSJ = VectorXf::Zero(encUnits);
178  auto dv_ = encPSJ.data();
179  for (int r = 0; r < encUnits; r++, ref_j++, dv_++){
180  float softmax = exp(-(pow((*ref_j)- v, 2.0))/sigma2);
181  *dv_ = softmax;
182  normalization += softmax;
183  }
184 
185  encPSJ/= normalization;
186  encData_t.push_back(encPSJ);
187 
188  }
189  output.push_back(encData_t);
190  }
191 }

193 void Dataset::softmax(float2DContainer& input, vectorXf2DContainer& output){
194 
195  float sigma2 = sigma*sigma;
196  int nT = input.size();
197 
198  for (int t = 0; t < nT; t++){
199 
200  float1DContainer d_t = input[t];
201  vectorXf1DContainer encUnits_j;
202 
203  for (int j = 0; j < nDof; j++){
204 
205  int encUnits = nUnits[j];
206 
207  VectorXf encPSJ = VectorXf::Zero(encUnits);
208 
209  float d_tj = d_t[j];
210 
211  vector<float> ref_j = ref[j];
212 
213  float v = (d_tj - jmin[j])/jrange[j];
214 
215  float normalization = 0.0;
216 
217  for (int r = 0; r < encUnits; r++){
218 
219  float softmax = exp(-(pow(ref_j[r]- v, 2.0))/sigma2);
220  encPSJ(r) = softmax;
221  normalization += softmax;
222  }
223 
224  // normalizing the distributions
225  for (int r = 0; r < encUnits; r++){
226  encPSJ(r) /= normalization;
227  }
228  encUnits_j.push_back(encPSJ);
229 
230  }
231  output.push_back(encUnits_j);
232  }
233 }
