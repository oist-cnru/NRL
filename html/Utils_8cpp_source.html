104 void Utils::getProperties(map<string,float1DContainer>& _mapFloat1D, map<string,string>& _mapString, map<string,bool>& _mapBool, string _path){
105 
106  string path = _path;
107  if (path.empty()){
108  path = string("config/properties.d");
109  }
110 
111  ifstream propFile(path);
112  if (!propFile.is_open())
113  throw Exception("The properties file could not be opened");
114 
115  string line;
116  std::string::size_type sz;
117  std::string token;
118  size_t pos = 0;
119  string delim1 = "=";
120 
121  //cout << "Configurations: "<< endl;
122  //cout << "------------------------" << endl;
123 
124  while (getline(propFile, line)){
125 
126  if (line.size() == 0)
127  continue;
128  pos = line.find(delim1);
129  string key = line.substr(0, pos);
130  trim(key);
131  tolower(key);
132 
133  if (key.at(0) == '#')
134  continue;
135 
136  //cout << line << endl;
137 
138  line.erase(0, pos + delim1.length());
139 
140  if (key == "modelpath"){
141  trim(line);
142  _mapString["modelpath"] = line;
143  continue;
144  }
145  else if (key == "datapath"){
146  trim(line);
147  _mapString["datapath"] = line;
148  continue;
149  }
150  else if (key == "network"){
151  trim(line);
152  _mapString["network"] = line;
153  continue;
154  }
155  else if (key == "robot"){
156  trim(line);
157  _mapString["robot"] = line;
158  continue;
159  }
160  else if (key == "shuffle"){
161  trim(line);
162  _mapBool["shuffle"] = (line == "true");
163  continue;
164  }
165  else if (key == "retrain"){
166  trim(line);
167  _mapBool["retrain"] = (line == "true");
168  continue;
169  }
170  else if (key == "greedy"){
171  trim(line);
172  _mapBool["greedy"] = (line == "true");
173  continue;
174  }
175 
176  float1DContainer value;
177 
178  while ((pos = line.find(delimiter)) != std::string::npos) {
179  token = line.substr(0, pos);
180  try{
181  value.push_back(std::stof(token,&sz));
182  }
183  catch (const std::invalid_argument& ia) {
184  std::cerr << "Invalid argument: " << ia.what() << '\n';
185  }catch (const std::out_of_range& oor) {
186  std::cerr << "Out of Range error: " << oor.what() << " token ["<< token << "]" << endl;
187  }
188  line.erase(0, pos + delimiter.length());
189  }
190 
191  try{
192  value.push_back(std::stof(line,&sz));
193  }catch (const std::invalid_argument& ia) {
194  std::cerr << "Invalid argument: " << ia.what() << '\n';
195  }catch (const std::out_of_range& oor) {
196  std::cerr << "Out of Range error: " << oor.what() << " token ["<< token << "]" << endl;
197  }
198  _mapFloat1D[key] = value;
199  }
200  //cout << "------------------------" << endl;
201 
202  propFile.close();
203 
204 }

211 MatrixXf Utils::kaiming_uniform_initialization(int _d0, int _d1, nonlinearity _nl){
212 
213  float gain = 0.0;
214  float negative_slope = sqrt(5.0);
215 
216  switch (_nl){
217  case nonlinearity::Linear:
218  case nonlinearity::Conv:
219  case nonlinearity::Sigmoid: gain = 1.0; break;
220  case nonlinearity::Tanh: gain = 5.0/3.0; break;
221  case nonlinearity::ReLU: gain = sqrt(2.0); break;
222  case nonlinearity::ReLU_Leaky: gain = sqrt(2.0 / (1 + (negative_slope*negative_slope))); break;
223  }
224 
225  float stdev = gain / sqrt(((float)_d1));
226 
227  // Calculating uniform bounds from standard deviation
228  float bound = sqrt(3.0) * stdev ;
229  MatrixXf W = MatrixXf::Random(_d0,_d1)*bound;
230  return W;
231 
232 }
