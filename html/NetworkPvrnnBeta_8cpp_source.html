55  NetworkPvrnnBeta::NetworkPvrnnBeta(map<string,float1DContainer>& _float1DMap, Dataset* _dataset){
56 
57  dataset = _dataset;
58 
59  if(_float1DMap.find("d") == _float1DMap.end()) throw Exception("'d' property not found");
60  float1DContainer fDNum = _float1DMap["d"];
61  for (unsigned int i = 0; i < fDNum.size(); i++){
62  int v = int(fDNum[i]);
63  if (v <= 0)
64  throw Exception("'d' property should include positive integer(s) greater than zero");
65  d_num.push_back(v);
66  }
67 
68  if(_float1DMap.find("z") == _float1DMap.end()) throw Exception("'z' property not found");
69  float1DContainer fZNum = _float1DMap["z"];
70  for (unsigned int i = 0; i < fZNum.size(); i++){
71  int v = int(fZNum[i]);
72  if (v <= 0)
73  throw Exception("'z' property should include positive integer(s) greater than zero");
74  z_num.push_back(v);
75  }
76 
77  if(_float1DMap.find("t") == _float1DMap.end()) throw Exception("'t' property not found");
78  float1DContainer fTau = _float1DMap["t"];
79  for (unsigned int i = 0; i < fTau.size(); i++){
80  int v = int(fTau[i]);
81  if (v <= 0)
82  throw Exception("'t' property should include positive integer(s) greater than zero");
83  tau.push_back(v);
84  }
85 
86  if(_float1DMap.find("w1") == _float1DMap.end()) throw Exception("'w1' property not found");
87  w1 = _float1DMap["w1"];
88  for (unsigned int i = 0; i < w.size(); i++){
89  if (w1[i] < 0)
90  throw Exception("'w1' property should include positive real number(s)");
91  }
92 
93  if(_float1DMap.find("w") == _float1DMap.end()) throw Exception("'w' property not found");
94  w = _float1DMap["w"];
95  for (unsigned int i = 0; i < w.size(); i++){
96  if (w[i] < 0)
97  throw Exception("'w' property should include positive real number(s)");
98  }
99 
100  layer_num = d_num.size();
101  ut = Utils::getInstance();
102 
103  int checksum = d_num.size() + z_num.size() + tau.size() + w.size();
104 
105  if (checksum != layer_num*4)
106  throw Exception("The network parameters are not correctly defined");
107 
108  dataset->getNunitsPerDim(o_num);
109  prim_num = _dataset->getNPrim();
110  prim_len = dataset->getPrimLength();
111 
112  int z_sum = 0;
113  for (unsigned int i = 0 ; i < z_num.size(); i++)
114  z_sum += z_num[i];
115 
116  o_dim = o_num.size();
117  rec_coef = 1.0/((float)(o_dim*1.0));
118  reg_coef = 1.0/((float)(z_sum*1.0));
119 
120  cout << "Number of internal layers: " << layer_num << endl;
121  state_dim = 0;
122  for (int l = 0; l < layer_num; l++){
123 
124  int d_num_top = (l==layer_num-1)? 0 : d_num[l+1];
125 
126  ILayer* layer = new LayerPvrnnBeta(l, d_num[l], d_num_top, z_num[l], z_sum, tau[l], (l < layer_num ? tau[l+1] : 0.0), prim_num, prim_len, w1[l], w[l]);
127  layers.push_back(layer);
128  state_dim += layer->getStateDim();
129 
130  }
131 
132  // connections to output layers
133  l0_d_num = d_num[0];
134  l0 = layers[0];
135  l0_context = static_cast<ContextPvrnnBeta*>(l0->getContext());
136 
137  for (int o = 0; o < o_dim ; o++){
138 
139  int num = o_num[o];
140  MatrixXf Wdx_ = ut->kaiming_uniform_initialization(num, l0_d_num, Utils::nonlinearity::Linear);
141  MatrixXf WdxT_ = Wdx_.transpose();
142  Wdo.push_back(Wdx_);
143  Wdo_transpose.push_back(WdxT_);
144  g_Wdo.push_back(MatrixXf::Zero(num, l0_d_num));
145  m_Wdo.push_back(MatrixXf::Zero(num, l0_d_num));
146  v_Wdo.push_back(MatrixXf::Zero(num, l0_d_num));
147  Bo.push_back(ut->kaiming_uniform_initialization(num));
148  g_Bo.push_back(VectorXf::Zero(num));
149  m_Bo.push_back(VectorXf::Zero(num));
150  v_Bo.push_back(VectorXf::Zero(num));
151 
152  }
153 
154  e_prim_id = 0;
155  e_cur_time = 0;
156  e_num_times = 0;
157  e_window_size = 0;
158  e_store_gen = false;
159  e_store_inference = false;;
160  }

244  void NetworkPvrnnBeta::t_backward(int _prim_id, vectorXf2DContainer& _X, vectorXf3DContainer& _Y, float& _rec, float& _reg, float& _loss){
245 
246  float1DContainer klDiv_l;
247  for (int l = 0; l < layer_num; l++){
248  klDiv_l.push_back(0.0);
249  }
250 
251  for (unsigned int s = 0; s < _Y.size(); s++){ // for all the batch samples
252 
253 
254  vectorXf2DContainer Ys = _Y[s];
255  for (int l = 0; l < layer_num; l++){
256  layers[l]->t_initBackward();
257  }
258 
259  int t_prev = prim_len-1;
260 
261  for (int t = prim_len; t > 0; t--, t_prev--){
262 
263  vectorXf1DContainer X_t = _X[t_prev];
264  vectorXf1DContainer Y_st = Ys[t_prev];
265  VectorXf g_dqloss = VectorXf::Zero(l0_d_num);
266  RowVectorXf L0_dq = l0_context->t_dq[_prim_id][t];
267 
268  for (int o = 0; o < o_dim; o++){
269 
270  ArrayXf Xpto = X_t[o];
271  ArrayXf Ypsto = Y_st[o];
272  ArrayXf yx = (Ypsto/Xpto) + NON_ZERO;
273  VectorXf recErr_t = Ypsto*(yx.log());
274  VectorXf gxloss_to = rec_coef*(Xpto-Ypsto);
275 
276  g_dqloss += Wdo_transpose[o]*gxloss_to ;
277 
278  g_Wdo[o] += gxloss_to*L0_dq;
279  g_Bo[o] += gxloss_to;
280 
281  _rec += recErr_t.sum();
282 
283  }
284 
285  ContextPvrnnBeta* prevC = nullptr;
286  for (int l = layer_num -1 ; l >= 0 ; l--){
287  ILayer* ll = layers[l];
288  ContextPvrnnBeta* lc = static_cast<ContextPvrnnBeta*>(ll->getContext());
289 
290  if (l == 0 ){
291  lc->g_dqloss = g_dqloss;
292  }
293  if (l < layer_num-1){
294  lc->g_hq_top = prevC->g_h_next;
295  lc->dq_top = prevC->t_dq[_prim_id][t];
296  }
297 
298  ll->t_backward(t, _prim_id);
299 
300  float wt = w[l];
301  if (t == 1)
302  wt = w1[l];
303  klDiv_l[l] += wt*lc->t_kld[_prim_id][t];
304 
305  prevC = lc;
306  }
307 
308  }
309  }
310 
311  for (int l = 0; l < layer_num; l++){
312  _reg += klDiv_l[l];
313  }
314  _loss = rec_coef*_rec + reg_coef*_reg;
315  }

318  void NetworkPvrnnBeta::t_optAdam(int _e, float _a, float _b1, float _b2){
319 
320  for (int o = 0; o < o_dim; o++){
321 
322  // updating parameters
323  ut->adam<MatrixXf>(&Wdo[o], &g_Wdo[o], &m_Wdo[o], &v_Wdo[o], _e, _a, _b1, _b2);
324  ut->adam<VectorXf>(&Bo[o], &g_Bo[o], &m_Bo[o], &v_Bo[o], _e, _a, _b1, _b2);
325 
326  Wdo_transpose[o] = Wdo[o].transpose();
327 
328  // clearing parameter gradients
329  ut->zero<MatrixXf>(&g_Wdo[o]);
330  ut->zero<VectorXf>(&g_Bo[o]);
331  }
332 
333  // updating the layer parameters
334  for (int l = 0; l < layer_num; l++){
335  layers[l]->t_optAdam(_e, _a, _b1, _b2);
336  }
337

341  float NetworkPvrnnBeta::getRecError(vectorXf2DContainer& _X, vectorXf3DContainer& _Y){
342 
343  float rec = 0.0;
344 
345  for (unsigned int s = 0; s < _Y.size(); s++){
346 
347  vectorXf2DContainer Ys = _Y[s];
348 
349  for (unsigned int t = 0; t < Ys.size(); t++){
350 
351  vectorXf1DContainer Yst = Ys[t];
352  vectorXf1DContainer Xt = _X[t];
353 
354  for (int o = 0; o < o_dim; o++){
355 
356  ArrayXf Xpto = Xt[o];
357  ArrayXf Ypsto = Yst[o];
358  ArrayXf yx = (Ypsto/Xpto) + NON_ZERO;
359  VectorXf recErr_t = Ypsto*(yx.log());
360  rec += recErr_t.sum();
361  }
362  }
363  }

655  ContextPvrnnBeta* lc = static_cast<ContextPvrnnBeta*>(ll->getContext());

659  }
660 
661  int t_prev = e_window_size-1;
662 
663  for (int t = e_window_size; t > 0; t--, t_prev--){
664 
665  vectorXf1DContainer X_t = _X[t_prev];
666  vectorXf1DContainer Y_t = _Y[t_prev];

668 
669  for (int o = 0; o < o_dim; o++){
670 
671  ArrayXf Xpto = X_t[o];
672  ArrayXf Ypsto = Y_t[o];
673  ArrayXf yx = (Ypsto/Xpto) + NON_ZERO;
674  VectorXf rec_t = Ypsto*(yx.log());
675  VectorXf gxloss_to = rec_coef*(Xpto-Ypsto);
676  g_dqloss += Wdo_transpose[o]*gxloss_to ;
677  _rec += rec_t.sum();
678  }
679 
680  ContextPvrnnBeta* prevC = nullptr;
681  for (int l = layer_num -1 ; l >= 0; l--){
682  ILayer* ll = layers[l];
683  ContextPvrnnBeta* lc = static_cast<ContextPvrnnBeta*>(ll->getContext());
684 
685  if (l == 0 ){
686  lc->g_dqloss = g_dqloss;
687  }
688  if (l < layer_num-1){
689  lc->g_hq_top = prevC->g_h_next;
690 
691  }
692 
693  ll->e_backward(t);
694 
695  kld_l[l] += *(kld_bw_i[l]++);
696  prevC = lc;
697  }
698 
699  }
700 
701  for (int l = 0; l < layer_num; l++){
702  _reg += w[l]*kld_l[l];
703  }
704  _loss = rec_coef*_rec + reg_coef*_reg;
705  }
