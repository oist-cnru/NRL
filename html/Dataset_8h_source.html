157  inline float Dataset::decodeSoftmax(ArrayXf& input, int dim){
158 
159  float1DContainer ref_ = ref[dim];
160 
161  auto d = input.data();
162  float1DContainer::iterator r =ref_.begin();
163 
164  float dec = 0.0;
165 
166  for (; r != ref_.end(); r++, d++){
167  dec += (*r)*(*d);
168  }
169  dec *= jrange[dim];
170  dec += jmin[dim];
171  return dec;
172 
173  }
