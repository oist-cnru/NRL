76 void demonstrateExperiment(LibNRL* nrl) {
77 
78  cout << endl << "simulation demonstration begin" << endl << endl;
79 
80  nrl->load();
81 
82 
83  int nDof = nrl->getNDof();
84 
85  if (nDof > 0) {
86 
87  int winSize = 15;
88  int winBufferSize = winSize * nDof;
89  CircularBuffer winBufferData(winSize, nDof);
90  int primId = 0;
91 
92  float* model_state = new float[nrl->getStateDim()];
93  for (int i = 0 ; i < nrl->getStateDim(); i++)
94  model_state[i] = 0.0;
95 
96  // The e_w parameters set bellow assume the network has two layers
97  // as in the original distribution of the sources
98  // in case more layers are set by changing the properties.d file,
99  // the same dimension for e_w must be considered
100  float e_w[2] = { 0.025, 0.025 };
101 
102  int expTimeSteps = 15;
103  int postdiction_epochs = 15;
104  float alpha = 0.1;
105  float beta1 = 0.9;
106  float beta2 = 0.999;
107  bool storeStates = false;
108  bool storeER = false;
109  bool showERLog = false;
110  nrl->e_enable(primId, winSize, e_w, expTimeSteps, postdiction_epochs,
111  alpha, beta1, beta2, storeStates, storeER);
112  float* tgt_pos = new float[nDof];
113  float* cur_pos = new float[nDof];
114 
115 
116  // information on free energy minimization
117  float elboOut[3] = { 0.0, 0.0, 0.0 };
118 
119  for (int i = 0; i < nDof; i++) {
120  tgt_pos[i] = 0.0;
121  cur_pos[i] = 0.0;
122  }
123 
124  int t = 0;
125 
126  bool endExperiment = false;
127 
128  while (!endExperiment) {
129 
130  // get time in ms
131  chrono::high_resolution_clock::time_point mst1 = chrono::high_resolution_clock::now();
132 
133  // < --- Here you should read the robot's joint state
134  // Since this is a dummy example, the current posture
135  // 'cur_pos' is set to zero
136  for (int i = 0; i < nDof; i++) {
137  cur_pos[i] = 0.0;
138  }
139 
140  // The target posture is generated by the RNN
141  nrl->e_generate(tgt_pos);
142 
143  // < --- Here you should call asynchronously the robot driver
144  // and send it tgt_pos to move the robot
145 
146  // store the current posture in the buffer
147  winBufferData.push(cur_pos);
148 
149  if (winBufferData.size() == winSize) {
150  t += 1;
151 
152  // call to postdiction (inference) process
153  // Optional: Information on free energy minimization (elboOut)
154  // can be obtained and analyzed on-line
155  nrl->e_postdict(winBufferData.getData(), elboOut, showERLog);
156 
157  // Optional: The latent state of the network can be obtained
158  // analyzed on-line or stored for future analysis
159  nrl->e_getState(model_state);
160 
161  }
162 
163  // get time in ms
164  chrono::high_resolution_clock::time_point mst2 = chrono::high_resolution_clock::now();
165  float detla_t = chrono::duration_cast<chrono::milliseconds>(mst2 - mst1).count();
166  mst1 = mst2;
167  if (t > 0) {
168  std::cout << "Step: " << t << " in " << detla_t << " ms" << std::endl;
169  }
170 
171  // For real applications you should set the program to sleep according to the
172  // desired loop period
173 
174  float period = 0.0; // loop period in ms
175  float sleepTime = period - detla_t;
176  if (sleepTime > 1.0)
177  std::this_thread::sleep_for(std::chrono::milliseconds((int) sleepTime));
178 
179  if (t == expTimeSteps)
180  endExperiment = true;
181 
182  }
183  // deallocating buffers
184  delete[] tgt_pos;
185  delete[] cur_pos;
186  delete[] model_state;
187  } else {
188  cout << endl << "Hint: the model path may be incorrect, or perhaps it requires to be trained !" << endl;
189  }
190 
191  cout << endl << "simulation demonstration end" << endl << endl;
192 }
