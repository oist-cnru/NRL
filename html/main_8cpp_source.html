77 void demonstrateExperiment(LibNRL* nrl) {
78 
79  cout << endl << "simulation demonstration begin" << endl << endl;
80 
81  nrl->load();
82 
83 
84  int nDof = nrl->getNDof();
85 
86  if (nDof > 0) {
87 
88  int winSize = 15;
89  int winBufferSize = winSize * nDof;
90  CircularBuffer winBufferData(winSize, nDof);
91  int primId = 0;
92 
93  float* model_state = new float[nrl->getStateDim()];
94  for (int i = 0 ; i < nrl->getStateDim(); i++)
95  model_state[i] = 0.0;
96 
97  // The e_w parameters set bellow assume the network has two layers
98  // as in the original distribution of the sources
99  // in case more layers are set by changing the properties.d file,
100  // the same dimension for e_w must be considered
101  float e_w[2] = { 0.025, 0.025 };
102 
103  int expTimeSteps = 15;
104  int postdiction_epochs = 15;
105  float alpha = 0.1;
106  float beta1 = 0.9;
107  float beta2 = 0.999;
108  bool storeStates = false;
109  bool storeER = false;
110  bool showERLog = false;
111  nrl->e_enable(primId, winSize, e_w, expTimeSteps, postdiction_epochs,
112  alpha, beta1, beta2, storeStates, storeER);
113  float* tgt_pos = new float[nDof];
114  float* cur_pos = new float[nDof];
115 
116 
117  // information on free energy minimization
118  float elboOut[3] = { 0.0, 0.0, 0.0 };
119 
120  for (int i = 0; i < nDof; i++) {
121  tgt_pos[i] = 0.0;
122  cur_pos[i] = 0.0;
123  }
124 
125  int t = 0;
126 
127  bool endExperiment = false;
128 
129  while (!endExperiment) {
130 
131  // get time in ms
132  chrono::high_resolution_clock::time_point mst1 = chrono::high_resolution_clock::now();
133 
134  // < --- Here you should read the robot's joint state
135  // Since this is a dummy example, the current posture
136  // 'cur_pos' is set to zero
137  for (int i = 0; i < nDof; i++) {
138  cur_pos[i] = 0.0;
139  }
140 
141  // The target posture is generated by the RNN
142  nrl->e_generate(tgt_pos);
143 
144  // < --- Here you should call asynchronously the robot driver
145  // and send it tgt_pos to move the robot
146 
147  // store the current posture in the buffer
148  winBufferData.push(cur_pos);
149 
150  if (winBufferData.size() == winSize) {
151  t += 1;
152 
153  // call to postdiction (inference) process
154  // Optional: Information on free energy minimization (elboOut)
155  // can be obtained and analyzed on-line
156  nrl->e_postdict(winBufferData.getData(), elboOut, showERLog);
157 
158  // Optional: The latent state of the network can be obtained
159  // analyzed on-line or stored for future analysis
160  nrl->e_getState(model_state);
161 
162  }
163 
164  // get time in ms
165  chrono::high_resolution_clock::time_point mst2 = chrono::high_resolution_clock::now();
166  float detla_t = chrono::duration_cast<chrono::milliseconds>(mst2 - mst1).count();
167  mst1 = mst2;
168  if (t > 0) {
169  std::cout << "Step: " << t << " in " << detla_t << " ms" << std::endl;
170  }
171 
172  // For real applications you should set the program to sleep according to the
173  // desired loop period
174 
175  float period = 0.0; // loop period in ms
176  float sleepTime = period - detla_t;
177  if (sleepTime > 1.0)
178  std::this_thread::sleep_for(std::chrono::milliseconds((int) sleepTime));
179 
180  if (t == expTimeSteps)
181  endExperiment = true;
182 
183  }
184  // deallocating buffers
185  delete[] tgt_pos;
186  delete[] cur_pos;
187  delete[] model_state;
188  } else {
189  cout << endl << "Hint: the model path may be incorrect, or perhaps it requires to be trained !" << endl;
190  }
191 
192  cout << endl << "simulation demonstration end" << endl << endl;
193 }
