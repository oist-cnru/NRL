290  void LayerPvrnnBeta::t_generate(int _time, int _prim_id){
291 
292  //generating from the prior distribution
293  VectorXf hp = t_hp[_prim_id].back();
294  VectorXf dp = c->t_dp[_prim_id].back();
295 
296  ArrayXf up;
297  ArrayXf lp;
298 
299  if (_time < gen_time_thres) {
300  up = Wduq*dp + Buq + t_au[_prim_id][_time];
301  lp = Wdlq*dp + Blq + t_al[_prim_id][_time];
302  }
303  else{
304  up = Wdup*dp + Bup;
305  lp = Wdlp*dp + Blp;
306  }
307 
308  ut->tanH<ArrayXf>(&up);
309  ArrayXf sp = lp.exp();
310  ArrayXf np = ArrayXf::Zero(z_num);
311  ut->randN<ArrayXf>(&np);
312  VectorXf zp = up + sp*np;
313 
314  hp = one_sub_eps*hp + eps*(Wdh*dp + Wzh*zp + Bh);
315 
316  if (!top)
317  hp += eps*(Wdh_top*c->dp_top);

319  dp = hp;

346  void LayerPvrnnBeta::t_forward(int _time, int _prim_id){
347 
348  // --------------- generation from the prior distribution ---------------
349 
350  VectorXf hp = t_hp[_prim_id].back();
351  VectorXf dp = c->t_dp[_prim_id].back();
352 
353  ArrayXf up;
354  ArrayXf lp;
355  if (_time == 0){
356  // unit Gaussian distribution
357  up = ArrayXf::Zero(z_num);
358  lp = ArrayXf::Zero(z_num);
359  }else{
360  up = Wdup*dp + Bup;
361  ut->tanH<ArrayXf>(&up);
362  lp = Wdlp*dp + Blp;
363  }
364  ArrayXf sp = lp.exp();
365  ArrayXf np = ArrayXf::Zero(z_num);
366  ut->randN<ArrayXf>(&np);
367  VectorXf zp = up + sp*np;
368 
369  hp = one_sub_eps*hp + eps*(Wdh*dp + Wzh*zp + Bh);
370 
371  // --------------- generation from the posterior distribution ---------------
372 
373  VectorXf hq = t_hq[_prim_id].back();
374  VectorXf dq = c->t_dq[_prim_id].back();
375 
376  ArrayXf uq = Wduq*dq + Buq + t_au[_prim_id][_time];
377  ut->tanH<ArrayXf>(&uq);
378  ArrayXf lq = Wdlq*dq + Blq + t_al[_prim_id][_time];
379  ArrayXf sq = lq.exp();
380  ArrayXf nq = ArrayXf::Zero(z_num);
381 
382  ut->randN<ArrayXf>(&nq);
383  VectorXf zq = uq + sq*nq;
384 
385  hq = one_sub_eps*hq + eps*(Wdh*dq + Wzh*zq + Bh);
386 
387  if (!top){
388  hp += eps*Wdh_top*c->dp_top;
389  hq += eps*Wdh_top*c->dq_top;
390  }
391 
392  dp = hp;
393  ut->tanH<VectorXf>(&dp);
394 
395  dq = hq;
396  ut->tanH<VectorXf>(&dq);
397 
398  float kld = get_kld(up, sp, uq, sq);
399 
400  t_hp[_prim_id].push_back(hp);
401  c->t_dp[_prim_id].push_back(dp);
402  t_up[_prim_id].push_back(up);
403  t_lp[_prim_id].push_back(lp);

433  void LayerPvrnnBeta::t_backward(int _time, int _prim_id){
434 
435  ArrayXf up_next = ArrayXf::Zero(z_num);
436  ArrayXf uq_next = ArrayXf::Zero(z_num);
437  if (_time < prim_len){
438  up_next = t_up[_prim_id][_time+1];
439  uq_next = t_uq[_prim_id][_time+1];
440  }
441 
442  ArrayXf up = t_up[_prim_id][_time];
443  ArrayXf sp = t_sp[_prim_id][_time];
444  ArrayXf sq = t_sq[_prim_id][_time];
445  ArrayXf uq = t_uq[_prim_id][_time];
446  ArrayXf nq = t_nq[_prim_id][_time];
447  ArrayXf dq = c->t_dq[_prim_id][_time];
448  RowVectorXf zq = t_zq[_prim_id][_time];
449  RowVectorXf dp_prev_transpose = c->t_dp[_prim_id][_time-1];
450  RowVectorXf dq_prev_transpose = c->t_dq[_prim_id][_time-1];
451 
452  ArrayXf up_pow_2 = up.pow(2.0);
453  ArrayXf sp_pow_2 = sp.pow(2.0) + NON_ZERO;
454  ArrayXf uq_pow_2 = uq.pow(2.0);
455  ArrayXf sq_pow_2 = sq.pow(2.0);
456 
457  VectorXf g_d = eps*c->g_h_next.transpose()*Wdh;
458 
459  if (bottom){
460  g_d += c->g_dqloss;
461  }
462 
463  if (!top){
464  g_d += ((VectorXf)(eps_top*c->g_hq_top.transpose()*Wdh_top_transpose)).transpose();
465  }
466 
467  RowVectorXf g_uptanh_transpose = g_up_next_transpose.array()*(1.0 - up_next.pow(2.0).transpose());
468  RowVectorXf g_uqtanh_transpose = g_uq_next_transpose.array()*(1.0 - uq_next.pow(2.0).transpose());
469 
470  g_d += g_uptanh_transpose*Wdup;
471  g_d += g_uqtanh_transpose*Wduq;
472  g_d += g_lp_next_transpose*Wdlp;
473  g_d += g_lq_next_transpose*Wdlq;
474 
475 
476  VectorXf g_h = g_d.array()*(1.0 - dq.pow(2.0)) + (one_sub_eps * c->g_h_next.array());
477  ArrayXf g_z = eps*g_h.transpose()*Wzh;
478 
479  RowVectorXf g_up;
480  RowVectorXf g_lp;
481  RowVectorXf g_uq;
482  RowVectorXf g_lq;
483 
484  float wFactor = w_div_z_sum;
485 
486  if (_time == 1)
487  wFactor = w1_div_z_sum;
488 
489  g_up = (wFactor*((up - uq)/sp_pow_2));
490  g_lp = wFactor*(1.0 - (((ArrayXf)(uq-up)).pow(2.0) + sq_pow_2)/sp_pow_2);
491  g_uq = (g_z + wFactor*((uq - up)/sp_pow_2));
492  g_lq = g_z*sq*nq + wFactor*(-1.0 + (sq_pow_2/sp_pow_2));
493 
494  // Parameter gradients
495 
496  g_Wdh += eps*g_h*dq_prev_transpose;
497  g_Bh += eps*g_h;
498 
499 
500  if (!top){
501  g_Wdh_top += eps * g_h * c->dq_top.transpose();
502  }
503 
504  g_Wzh += eps* g_h* zq;
505 
506  VectorXf g_uqtanh = g_uq.array().transpose()*(1.0 - uq_pow_2);
507 
508  g_Wduq += g_uqtanh*dq_prev_transpose;
509  g_Buq += g_uqtanh;
510  g_Wdlq += g_lq.transpose()*dq_prev_transpose;
511  g_Blq += g_lq;
512 
513  VectorXf g_uptanh = g_up.array().transpose()*(1.0 - up_pow_2);
514 
515  g_Wdup += g_uptanh*dp_prev_transpose;
516  g_Bup += g_uptanh;
517  g_Wdlp += g_lp.transpose()*dp_prev_transpose;
518  g_Blp += g_lp;
519 
520  t_g_au[_prim_id][_time-1] = g_uqtanh;
521  t_g_al[_prim_id][_time-1] = g_lq;
522 
523  c->g_h_next = g_h;
524  g_up_next_transpose = g_up;
525  g_uq_next_transpose = g_uq;
526  g_lp_next_transpose = g_lp;
527  g_lq_next_transpose = g_lq;
528 
529  }
